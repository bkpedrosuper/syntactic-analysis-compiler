import math;

func main() {
	let n = stdin().parser().int();
	let numberOfPrimes = 0;

	for i:(0,n,1) {
		if isPrime(i) {
			numberOfPrimes = numberOfPrimes + 1;
		}
	}

	println("O número de números primos entre 0 e ", n, "É igual a ", numberOfPrimes);
}

func isPrime(number) {
	let cont = 0;

	for i:(1, math.sqrt(number), 1) {
		if i mod 2 == 0 {
			cont = cont + 1;
		}
	}

	return cont == 2;
}
